def import_munc(munc_file_path):
    munc_model = onnx_model.ONNXModel(munc_file_path)
    mi = inference_torch
    munc_graph = graph_ops.GraphOps(munc_model, stats_collector_obj=None, hwconfig=hw_config.boreasA_pfsr8)
    return munc_model, munc_graph
    

if REBUILD_RESNET18_ONNX:
    # save resnet18 graph for bcm
    bcm_munc_model, bcm_munc_graph = import_munc(munc_file_path)
    bcm_utils.retrain2bcm(bcm_munc_graph, bcm_mma=BCM_MMA, mma_attr=MMA_ATTRS)
    bcm_munc_model.save(os.path.join(ROOT_DIR, 'resnet18_bcm.onnx'))

def plot_comp(bcm_data, orig_data):
    plt.figure()
    plt.plot(orig_data.ravel(), bcm_data.ravel(), '.')
    plt.ylabel('BCM output')
    plt.xlabel('Munc output')
    plt.plot([-255, 255], [-255, 255], '--')
    plt.xlim(-255, 255)
    plt.ylim(-255, 255)
    plt.gca().set_aspect('equal', adjustable='box')
    plt.figure()
    plt.axhline(-1, color='r', ls='--')
    plt.axhline(1, color='r', ls='--')
    plt.axhline(0, color='g', ls='--')
    plt.plot(orig_data.ravel(), bcm_data.ravel() - orig_data.ravel(), '.')
    plt.xlabel('Munc output')
    plt.ylabel('error')
    plt.figure()
    plt.hist(np.ravel(bcm_data - orig_data), bins=200);
    plt.xlabel('error')
    mse = np.mean(np.ravel(bcm_data - orig_data)**2)
    print(f"MSE: {mse:3f}")
